rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Subtitles collection
    match /subtitles/{subtitleId} {
      // Anyone can read subtitles
      allow read: if true;
      
      // Only authenticated users can create subtitles
      // And they must set themselves as the uploader
      allow create: if request.auth != null 
                    && request.resource.data.uploaderId == request.auth.uid
                    && request.resource.data.uploaderName is string
                    && request.resource.data.title is string
                    && request.resource.data.fileUrl is string;
      
      // Only the uploader can update or delete their subtitles
      allow update, delete: if request.auth != null 
                            && resource.data.uploaderId == request.auth.uid;
    }
    
    // Future: User profiles collection
    match /users/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
